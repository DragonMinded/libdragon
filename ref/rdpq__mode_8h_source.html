387inline void rdpq_mode_antialias(rdpq_antialias_t mode)
388{
389 // Just enable/disable SOM_AA_ENABLE. The RSP will then update the render mode
390 // which would trigger different other bits in SOM depending on the current mode.
391 __rdpq_mode_change_som(SOM_AA_ENABLE | SOMX_AA_REDUCED,
392 (mode ? SOM_AA_ENABLE : 0) | (mode == AA_REDUCED ? SOMX_AA_REDUCED : 0));

465inline void rdpq_mode_combiner(rdpq_combiner_t comb) {
466 extern void __rdpq_fixup_mode(uint32_t cmd_id, uint32_t w0, uint32_t w1);
467 extern void __rdpq_fixup_mode4(uint32_t cmd_id, uint32_t w0, uint32_t w1, uint32_t w2, uint32_t w3);
468
469 if (comb & RDPQ_COMBINER_2PASS)
470 __rdpq_fixup_mode(RDPQ_CMD_SET_COMBINE_MODE_2PASS,
471 (comb >> 32) & 0x00FFFFFF,
472 comb & 0xFFFFFFFF);
473 else {
474 rdpq_combiner_t comb1_mask = RDPQ_COMB1_MASK;
475 if (((comb >> 0 ) & 7) == 1) comb1_mask ^= 1ull << 0;
476 if (((comb >> 3 ) & 7) == 1) comb1_mask ^= 1ull << 3;
477 if (((comb >> 6 ) & 7) == 1) comb1_mask ^= 1ull << 6;
478 if (((comb >> 18) & 7) == 1) comb1_mask ^= 1ull << 18;
479 if (((comb >> 21) & 7) == 1) comb1_mask ^= 1ull << 21;
480 if (((comb >> 24) & 7) == 1) comb1_mask ^= 1ull << 24;
481 if (((comb >> 32) & 31) == 1) comb1_mask ^= 1ull << 32;
482 if (((comb >> 37) & 15) == 1) comb1_mask ^= 1ull << 37;
483
484 __rdpq_fixup_mode4(RDPQ_CMD_SET_COMBINE_MODE_1PASS,
485 (comb >> 32) & 0x00FFFFFF,
486 comb & 0xFFFFFFFF,
487 (comb1_mask >> 32) & 0x00FFFFFF,
488 comb1_mask & 0xFFFFFFFF);
489 }
490}

874inline void __rdpq_mode_change_som(uint64_t mask, uint64_t val)
875{
876 // This is identical to #rdpq_change_other_modes_raw, but we also
877 // set bit 1<<15 in the first word. That flag tells the RSP code
878 // to recalculate the render mode, in addition to flipping the bits.
879 // #rdpq_change_other_modes_raw instead just changes the bits as
880 // you would expect from a raw API.
881 extern void __rdpq_fixup_mode3(uint32_t cmd_id, uint32_t w0, uint32_t w1, uint32_t w2);
882 if (mask >> 32)
883 __rdpq_fixup_mode3(RDPQ_CMD_MODIFY_OTHER_MODES, 0 | (1<<15), ~(mask >> 32), val >> 32);
884 if ((uint32_t)mask)
885 __rdpq_fixup_mode3(RDPQ_CMD_MODIFY_OTHER_MODES, 4 | (1<<15), ~(uint32_t)mask, (uint32_t)val);
886}
